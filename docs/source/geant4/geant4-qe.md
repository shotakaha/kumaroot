G4double QuantumEfficiency(G4double energy)
{
  G4double qe = 0.;
  G4double trans = 0.;

  // QE part
  if( energy<PhotonEnergyQE[0] ||
      energy>PhotonEnergyQE[nEntriesQE-1] )
    qe = 0.;

  else {
    for(int i=0; i<(nEntriesQE-1); i++) {
      if( energy>=PhotonEnergyQE[i] &&
          energy<PhotonEnergyQE[i+1] ) {
        qe = QE[i];
        break;
      }
    }
  }

  //G4cout << "QE:" << qe << G4endl;

  // Transmittion part
  if( energy<PhotonEnergyT[0] )
    trans = 1.;

  else if( energy>PhotonEnergyT[nEntriesT-1] )
    trans = 0.;

  else {
    for(int i=0; i<(nEntriesT-1); i++) {
      if( energy>=PhotonEnergyT[i] &&
          energy<PhotonEnergyT[i+1] ) {
        trans = Trans[i];
        break;
      }
    }
  }

  //G4cout << "Trans:" << trans << G4endl;


  return (qe*trans);

}
